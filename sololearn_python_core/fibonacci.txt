n = int(input())


def fibonacci(n):
    if n == 1:
        res = 0
        print(res)
    elif n == 2:
        res = [0, 1]
        print(res) 
    else:
        res = [0, 1]
        for i in range(n-2):
            res.append(res[-1]+res[-2])
        print(*res, sep ='\n')

fibonacci(n)
